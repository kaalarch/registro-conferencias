CREATE TABLE `conferencias`.`1` 
( `FOLIO` INT(10) NOT NULL AUTO_INCREMENT , 
`NOM_CONFE` VARCHAR(100) NOT NULL , 
`SEDE` VARCHAR(100) NOT NULL , 
`FCH_CONFE` DATE NOT NULL , 
`HRA_INI_CONFE` TIME NOT NULL , 
`HRA_FIN_CONFE` TIME NOT NULL , 
`NOM_CONFERENCISTA` VARCHAR(50) NOT NULL , 
`AP_CONFERENCISTA` VARCHAR(50) NOT NULL , 
`AM_CONFERENCISTA` VARCHAR(50) NULL , 
`NO_LUGARES` INT(6) NOT NULL , 
PRIMARY KEY (`FOLIO`)) 
ENGINE = InnoDB;
